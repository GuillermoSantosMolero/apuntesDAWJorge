
TABLA PAIS:
	ATRIBUTOS:
		IDPAIS,NOMBRE
	RESTRICCIONES:
		-PRIMARY KEY: IDPAIS;

TABLA PROVINCIA
	ATRIBUTOS:
		IDPROVICIA,PAIS,NOMBRE
	RESTRICCIONES:
		-PRIMARY KEY: IDPROVINCIA.
		-FOREIGN KEY: PAIS REFERENCIA A IDPAIS DE LA TABLA PAIS.
TABLA IDIOMA:
	ATRIBUTOS:
		IDIDIOMA,DESCRIPCION.
	RESTRICCIONES:
		-PRIMARY KEY: IDIDIOMA.

TABLA USER:
	Atributos:
		IDUSER,NOMBRE,APELLIDOS,CORREO,FECHA DE NACIMIENTO,SEXO,TELEFONO,IDIOMA,DATOSADICIONALES,PAIS,
		PROVINCIA,CIUDAD,DIRECCION,CODIGOPOSTAL.
	Restricciones:
		-PRIMARY KEY: IDUSER.
		-UNIQUE KEY CORREO.
		-FOREIGN KEY: PAIS REFERENCIA A IDPAIS DE LA TABLA PAIS.
		-FOREIGN KEY: PROVICIA REFERENCIA A IDPROVINCIA DE LA TABLA PROVINCIA.
		-FOREIGN KEY: IDIOMA REFERENCIA A IDIDIOMA DE LA TABLA IDIOMA.
